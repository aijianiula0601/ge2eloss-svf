[program: asr2text]

user = huangjiahong ;
autostart=true ;
autorestart=true ;
environment = PYTHONPATH="/home/huangjiahong/svn_projects/asr_keras" ;
redirect_stderr=true ;
logfile_backups=10 ;
stdout_logfile_backups=10 ;
logfile_maxbytes=50MB ;
stderr_logfile=/home/huangjiahong/tmp/logs/api/asr/err.log ;
stdout_logfile=/home/huangjiahong/tmp/logs/api/asr/info.log ;
